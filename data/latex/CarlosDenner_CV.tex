% Title formatting: bold and clickable if DOI exists

  \iffieldundef{doi}
    {\textbf{#1}}

  \iffieldundef{doi}
    {\textbf{#1}}
    {\href{https://doi.org/\thefield{doi}}{\textbf{#1}}}}

  \iffieldundef{doi}
    {\textbf{#1}}

% Volume in bold
\DeclareFieldFormat[article]{volume}{\textbf{#1}}

% Issue in parentheses
\DeclareFieldFormat[article]{number}{(#1)}

       \printfield{extradate}}}}

% Custom driver for articles: Year. Authors. Title. Journal Volume(Issue), Pages.
\DeclareBibliographyDriver{article}{%
  \usebibmacro{bibindex}%
  \usebibmacro{begentry}%
  % Year first (bold)
  \printtext[\textbf]{\printfield{labelyear}}%
  \setunit{\addperiod\space}%
  % Authors
  \printnames{author}%
  \setunit{\addperiod\space}%
  % Title (bold, clickable with DOI)
  \printfield{title}%
  \setunit{\addperiod\space}%
  % Journal name (italic)
  \printfield{journaltitle}%
  \setunit{\addspace}%
  % Volume (bold)
  \printfield{volume}%
  % Issue (in parentheses)
  \printfield{number}%
  \setunit{\addcomma\space}%
  % Pages
  \printfield{pages}%
  \setunit{\addperiod}%
  \usebibmacro{finentry}}

% Custom driver for book chapters
\DeclareBibliographyDriver{incollection}{%
  \usebibmacro{bibindex}%
  \usebibmacro{begentry}%
  \printtext[\textbf]{\printfield{labelyear}}%
  \setunit{\addperiod\space}%
  \printnames{author}%
  \setunit{\addperiod\space}%
  \printfield{title}%
  \setunit{\addperiod\space}%
  \printtext{In: }\printfield{booktitle}%
  \setunit{\addcomma\space}%
  \printfield{pages}%
  \setunit{\addperiod}%
  \usebibmacro{finentry}}

% Custom driver for @misc (datasets, etc.)
\DeclareBibliographyDriver{misc}{%
  \usebibmacro{bibindex}%
  \usebibmacro{begentry}%
  \printtext[\textbf]{\printfield{labelyear}}%
  \setunit{\addperiod\space}%
  \printnames{author}%
  \setunit{\addperiod\space}%
  \printfield{title}%
  \setunit{\addperiod\space}%
  \printfield{howpublished}%
  \setunit{\addperiod\space}%
  \printfield{note}%
  \setunit{\addperiod}%
  \usebibmacro{finentry}}
